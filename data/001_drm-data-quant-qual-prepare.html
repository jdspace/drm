read_and_standardize <- function(file) {
  # Read full CSV file
  df <- read.csv(file, fileEncoding = "UTF-8-BOM", stringsAsFactors = FALSE)

  # Save row 1 as variable names, row 2 as question text
  question_keys <- data.frame(
    variable = names(df),
    question = as.character(df[1, ]),
    stringsAsFactors = FALSE
  )

  # Remove row 2 from data
  df <- df[-1, ]

  # Detect "sona" column (case-insensitive)
  sona_col <- names(df)[str_detect(tolower(names(df)), "sona")]
  if (length(sona_col) == 0) {
    stop(paste("No SONA column found in", basename(file)))
  }

  # Rename to "SONA"
  df <- df %>% rename(SONA = all_of(sona_col[1]))

  return(list(data = df, question_keys = question_keys))
}

  1. Only keep the row of entry of each SONA with the most complete entry (fewer NA values)
# Create a named list: file -> duplicated SONA IDs
dup_sona_list <- split(duplicates_summary$SONA, duplicates_summary$file)

# Clean each data frame
data_list_cleaned <- map2(data_list, basename(file_paths), function(df, fname) {
  if (fname %in% names(dup_sona_list)) {
    dup_ids <- dup_sona_list[[fname]]

    df_deduped <- df %>%
      rowwise() %>%
      mutate(non_missing_count = sum(!is.na(c_across(everything())))) %>%
      group_by(SONA) %>%
      mutate(dup_flag = SONA %in% dup_ids) %>%
      filter(!dup_flag | (dup_flag & rank(-non_missing_count, ties.method = "first") == 1)) %>%
      ungroup() %>%
      select(-non_missing_count, -dup_flag)

    return(df_deduped)
  } else {
    return(df)
  }
})
